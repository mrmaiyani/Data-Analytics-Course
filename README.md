```python
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load & inspect
df = pd.read_csv('sales_data.csv', parse_dates=['order_date'])
print(df.info())
print(df.describe())

# Feature engineering
df['revenue']       = df['units'] * df['price']
df['month']         = df['order_date'].dt.to_period('M')
df['is_high_value'] = df['revenue'] > df['revenue'].quantile(0.75)

# Monthly revenue trend
monthly = df.groupby('month')['revenue'].sum().reset_index()
monthly['month'] = monthly['month'].astype(str)

fig, axes = plt.subplots(1, 2, figsize=(14, 5))
axes[0].bar(monthly['month'], monthly['revenue'], color='#378ADD', alpha=0.85)
axes[0].set_title('Monthly Revenue', fontsize=14, fontweight='bold')
axes[0].tick_params(axis='x', rotation=45)

sns.histplot(df['revenue'], bins=30, kde=True, color='#1D9E75', ax=axes[1])
axes[1].set_title('Revenue Distribution', fontsize=14, fontweight='bold')

plt.tight_layout()
plt.savefig('eda_output.png', dpi=150)
plt.show()
```

<summary>🗄️ SQL — Window Functions & CTE Analysis</summary>

```sql
-- Top regions by revenue with running total (last 90 days)
WITH regional_sales AS (
    SELECT
        region,
        product_category,
        SUM(revenue)                                          AS total_revenue,
        COUNT(DISTINCT order_id)                              AS total_orders,
        ROUND(AVG(revenue), 2)                                AS avg_order_value,
        RANK()     OVER (ORDER BY SUM(revenue) DESC)          AS revenue_rank,
        SUM(SUM(revenue)) OVER (ORDER BY SUM(revenue) DESC)  AS running_total
    FROM sales
    WHERE order_date >= CURRENT_DATE - INTERVAL '90 days'
      AND status = 'completed'
    GROUP BY region, product_category
)
SELECT
    region,
    total_revenue,
    total_orders,
    avg_order_value,
    revenue_rank,
    ROUND(100.0 * total_revenue / running_total, 1) AS pct_of_total
FROM regional_sales
WHERE revenue_rank <= 5
ORDER BY revenue_rank;
```

<summary>📊 DAX — Power BI Custom Measures</summary>

```dax
-- YoY Revenue Growth %
YoY Growth % =
VAR CurrentRevenue = [Total Revenue]
VAR PriorRevenue   =
    CALCULATE(
        [Total Revenue],
        SAMEPERIODLASTYEAR('Date'[Date])
    )
RETURN
    IF(
        NOT ISBLANK(PriorRevenue),
        DIVIDE(CurrentRevenue - PriorRevenue, PriorRevenue, 0),
        BLANK()
    )

-- Rolling 3-Month Average
Rolling 3M Avg =
AVERAGEX(
    DATESINPERIOD('Date'[Date], LASTDATE('Date'[Date]), -3, MONTH),
    [Total Revenue]
)
```
